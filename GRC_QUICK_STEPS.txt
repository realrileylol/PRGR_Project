   sys.path.append('/home/user/PRGR_Project')
   from kld2_sensor import KLD2Sensor

   class blk(gr.sync_block):
       def __init__(self):
           gr.sync_block.__init__(
               self,
               name='KLD2 Source',
               in_sig=None,
               out_sig=[np.float32]
           )
           self.sensor = KLD2Sensor(sample_rate=10000)
           self.sensor.start_stream()

       def work(self, input_items, output_items):
           samples = self.sensor.get_samples(len(output_items[0]))
           output_items[0][:] = ((samples - 1.65) / 1.65).astype(np.float32)
           return len(output_items[0])
